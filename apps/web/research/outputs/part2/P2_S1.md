```typescript
// Main orchestrator implementing S3's decision tree
export const classifyCorpusEntity = inngest.createFunction(
  {
    id: "corpus-classify-entity",
    concurrency: [{ limit: 10 }],
    retries: 3,
  },
  { event: "corpus/entity.classify" },
  async ({ event, step }) => {
    const { entity_id, source_type, jurisdiction = "federal" } = event.data;
    
    // Load jurisdiction-scoped rules
    const ruleSet = await step.run("load-rules", async () => {
      return loadRuleSet(jurisdiction);
    });
    
    // Initialize stage output
    let stageOutput: StageOutput = {
      entity_id,
      stage: "structural",
      score_components: { A: 0, K: 0, D: 0, B: 0 },
      composite_score: 0,
      classifications: [],
      s3_tier: "pending",
      needs_review: false,
      partial_signals: [],
    };
    
    // S3 Step 5: openFDA branch (bypass composite scoring)
    if (source_type === "openfda") {
      stageOutput = await step.run("openfda-classify", async () => {
        return classifyOpenFDA(entity_id, ruleSet);
      });
      return await persistClassification(stageOutput);
    }
    
    // S3 Step 1: CFR reference check (Stage 1)
    stageOutput = await step.run("stage-1-structural", async () => {
      return stage1Structural(entity_id, ruleSet, stageOutput);
    });
    
    // If Stage 1 resolved with high confidence, done
    if (stageOutput.composite_score >= 1.0 && stageOutput.classifications.length > 0) {
      return await persistClassification(stageOutput);
    }
    
    // S3 Steps 2-3 partial: Agency + document type (Stage 2)
    stageOutput = await step.run("stage-2-metadata", async () => {
      return stage2Metadata(entity_id, ruleSet, stageOutput);
    });
    
    // Early resolution check
    const earlyScore = stageOutput.score_components.A + 
                      stageOutput.score_components.D + 
                      stageOutput.score_components.B;
    
    if (earlyScore >= 0.60 && !stageOutput.partial_signals.some(s => s.type === "excluded_agency")) {
      stageOutput.composite_score = earlyScore;
      stageOutput.s3_tier = "relevant";
      return await persistClassification(stageOutput);
    }
    
    // S3 Step 3 complete: Keywords (Stage 3a)
    stageOutput = await step.run("stage-3a-keywords", async () => {
      return stage3aKeywords(entity_id, ruleSet, stageOutput);
    });
    
    // Apply S3 Step 4 thresholds
    stageOutput = applyS3Thresholds(stageOutput);
    
    // Route based on tier
    switch (stageOutput.s3_tier) {
      case "relevant":
        return await persistClassification(stageOutput);
        
      case "review":
        await step.run("stage-4-ai", async () => {
          return triggerAIClassification(entity_id, stageOutput);
        });
        break;
        
      case "irrelevant":
        await step.run("stage-5-confirm", async () => {
          return confirmIrrelevance(entity_id, stageOutput);
        });
        break;
        
      case "low_priority":
        return await persistClassification(stageOutput);
    }
  }
);

// Transition logic implementing S3 thresholds
export function applyS3Thresholds(output: StageOutput): StageOutput {
  const { composite_score } = output;
  
  // S3 Rule 2: Excluded agency override
  if (output.partial_signals.some(s => s.type === "excluded_agency")) {
    const K = output.score_components.K;
    const D = output.score_components.D;
    if (K >= 0.40 && D >= 0.10) {
      output.s3_tier = "review";
      output.needs_review = true;
      output.review_reason = "Excluded agency with strong keyword signal";
    } else {
      output.s3_tier = "irrelevant";
    }
    return output;
  }
  
  // S3 Step 4 thresholds
  if (composite_score >= 0.60) {
    output.s3_tier = "relevant";
  } else if (composite_score >= 0.40) {
    output.s3_tier = "review";
    output.needs_review = true;
    output.review_reason = "Borderline score requires AI classification";
  } else if (composite_score >= 0.20) {
    output.s3_tier = "low_priority";
  } else {
    output.s3_tier = "irrelevant";
  }
  
  return output;
}
```

### 2A. CFR Citation Parser

```typescript
// Generic interface for any jurisdiction
interface CitationParser {
  jurisdiction: string;
  parse(raw: string): ParsedCitation | null;
}

// Federal implementation
class FederalCitationParser implements CitationParser {
  jurisdiction = 'federal';
  
  // Patterns ordered by specificity
  private patterns = [
    // Full section reference: 21 CFR 1306.04(a)
    /(?:Title\s+)?(\d+)\s+CFR\s+(?:Part\s+)?(\d+)\.(\d+)(?:\([a-z]\))?/i,
    // Part reference: 21 CFR Part 1306 or 21 CFR 1306
    /(?:Title\s+)?(\d+)\s+CFR\s+(?:Part\s+)?(\d+)/i,
    // Section symbol: 21 CFR §1306.04
    /(?:Title\s+)?(\d+)\s+CFR\s+§\s*(\d+)\.(\d+)/i,
    // Verbose: Title 21, Chapter II, Part 1306
    /Title\s+(\d+),\s+Chapter\s+[IVXLC]+,\s+Part\s+(\d+)/i,
  ];
  
  parse(raw: string): ParsedCitation | null {
    for (const pattern of this.patterns) {
      const match = raw.match(pattern);
      if (match) {
        return {
          jurisdiction: 'federal',
          title: parseInt(match[1]),
          part: parseInt(match[2]),
          section: match[3] ? parseInt(match[3]) : undefined
        };
      }
    }
    return null;
  }
}
```

### 2B. Lookup Algorithm Using S2-F

```typescript
async function lookupStructuralClassification(
  citation: ParsedCitation,
  ruleSet: RuleSet
): Promise<DomainAssignment[]> {
  // 1. Find matching structural rules
  const rules = ruleSet.structural_rules.filter(rule => 
    rule.matcher.match(citation)
  );
  
  if (rules.length === 0) {
    // Check if title is relevant but part is not
    if (citation.jurisdiction === 'federal' && citation.title) {
      const relevantTitles = [2, 10, 16, 20, 21, 26, 28, 29, 32, 38, 40, 42, 45, 47, 49];
      if (relevantTitles.includes(citation.title)) {
        // Title is relevant but part not in allowlist
        return [{
          domain_code: 'uncategorized',
          confidence: 0.2,
          is_primary: true,
          assigned_by: 'structural',
          notes: 'Relevant title but part not in allowlist'
        }];
      }
    }
    return [];
  }
  
  // 2. Apply rules in priority order
  const assignments: DomainAssignment[] = [];
  const seenDomains = new Set<string>();
  
  for (const rule of rules) {
    if (!seenDomains.has(rule.domain_code)) {
      assignments.push({
        domain_code: rule.domain_code,
        confidence: rule.confidence,
        is_primary: assignments.length === 0,
        assigned_by: 'structural',
        rule_id: rule.rule_id,
        ai_refinement_needed: rule.ai_refinement_needed
      });
      seenDomains.add(rule.domain_code);
    }
  }
  
  return assignments;
}

// CFR matcher implementation
class CFRMatcher implements CitationMatcher {
  constructor(
    private title: number,
    private part: number,
    private sections?: number[]
  ) {}
  
  match(citation: ParsedCitation): boolean {
    if (citation.jurisdiction !== 'federal') return false;
    if (citation.title !== this.title) return false;
    if (citation.part !== this.part) return false;
    
    // If sections specified, must match one
    if (this.sections && citation.section) {
      return this.sections.includes(citation.section);
    }
    
    return true;
  }
}
```

### 2C. Federal Register CFR Inheritance

```typescript
async function extractFRCitations(entity: FederalRegisterEntity): Promise<ParsedCitation[]> {
  const citations: ParsedCitation[] = [];
  const parser = new FederalCitationParser();
  
  // Primary source: cfr_references field
  if (entity.cfr_references && Array.isArray(entity.cfr_references)) {
    for (const ref of entity.cfr_references) {
      const parsed = parser.parse(ref);
      if (parsed) citations.push(parsed);
    }
  }
  
  // Fallback: parse from title and abstract
  const textToSearch = [
    entity.title,
    entity.abstract,
    entity.action
  ].filter(Boolean).join(' ');
  
  // Extract all CFR references from text
  const cfrPattern = /\d+\s+CFR\s+(?:Part\s+)?\d+(?:\.\d+)?/gi;
  const matches = textToSearch.match(cfrPattern) || [];
  
  for (const match of matches) {
    const parsed = parser.parse(match);
    if (parsed && !citations.some(c => 
      c.title === parsed.title && 
      c.part === parsed.part &&
      c.section === parsed.section
    )) {
      citations.push(parsed);
    }
  }
  
  return citations;
}

async function classifyFRByStructural(
  entity: FederalRegisterEntity,
  ruleSet: RuleSet
): Promise<DomainAssignment[]> {
  const citations = await extractFRCitations(entity);
  if (citations.length === 0) return [];
  
  // Collect all domain assignments
  const domainCounts = new Map<string, number>();
  const allAssignments: DomainAssignment[] = [];
  
  for (const citation of citations) {
    const assignments = await lookupStructuralClassification(citation, ruleSet);
    for (const assignment of assignments) {
      allAssignments.push(assignment);
      const count = domainCounts.get(assignment.domain_code) || 0;
      domainCounts.set(assignment.domain_code, count + 1);
    }
  }
  
  // Primary domain = most referenced
  const sortedDomains = Array.from(domainCounts.entries())
    .sort((a, b) => b[1] - a[1]);
  
  // Deduplicate and mark primary
  const finalAssignments: DomainAssignment[] = [];
  for (let i = 0; i < sortedDomains.length; i++) {
    const [domain_code] = sortedDomains[i];
    const assignment = allAssignments.find(a => a.domain_code === domain_code)!;
    finalAssignments.push({
      ...assignment,
      is_primary: i === 0
    });
  }
  
  return finalAssignments;
}
```

```typescript
async function classifyOpenFDA(
  entity_id: string,
  ruleSet: RuleSet
): Promise<StageOutput> {
  const entity = await loadOpenFDAEntity(entity_id);
  
  // Find matching dataset rule
  const datasetRule = ruleSet.dataset_rules.find(rule =>
    entity.dataset_endpoint === rule.dataset_endpoint
  );
  
  if (!datasetRule) {
    return {
      entity_id,
      stage: 'structural',
      score_components: { A: 0.45, K: 0, D: 0, B: 0 },
      composite_score: 0.45,
      classifications: [],
      s3_tier: 'review',
      needs_review: true,
      review_reason: 'Unknown openFDA dataset',
      partial_signals: []
    };
  }
  
  // Apply field filters
  let matches = true;
  for (const filter of datasetRule.field_filters) {
    const fieldValue = getNestedField(entity, filter.field_path);
    
    switch (filter.operator) {
      case 'equals':
        matches = fieldValue === filter.value;
        break;
      case 'contains':
        matches = String(fieldValue).toLowerCase().includes(String(filter.value).toLowerCase());
        break;
      case 'in':
        matches = Array.isArray(filter.value) && filter.value.includes(fieldValue);
        break;
      case 'regex':
        matches = new RegExp(String(filter.value)).test(String(fieldValue));
        break;
    }
    
    if (!matches) break;
  }
  
  return {
    entity_id,
    stage: 'structural',
    score_components: { A: 0.45, K: 0, D: 0, B: 0 },  // A always 0.45 for FDA
    composite_score: 0.45,
    classifications: matches ? [{
      domain_code: datasetRule.domain_code,
      confidence: 0.9,
      is_primary: true,
      assigned_by: 'dataset',
      rule_id: datasetRule.rule_id
    }] : [],
    s3_tier: matches ? 'relevant' : 'review',
    needs_review: !matches,
    review_reason: !matches ? 'Field filters did not match' : undefined,
    partial_signals: []
  };
}
```

### 3B. Document Type Scoring (D Score)

```typescript
function calculateDocumentTypeScore(
  type: 'RULE' | 'PRORULE' | 'NOTICE' | 'PRESDOCU',
  action: string
): number {
  // Normalize action text for matching
  const normalizedAction = action.toLowerCase();
  
  switch (type) {
    case 'RULE':
      if (normalizedAction.includes('final rule')) {
        if (normalizedAction.includes('interim')) return 0.15;  // Interim final
        if (normalizedAction.includes('direct')) return 0.15;   // Direct final
        if (normalizedAction.includes('temporary')) return 0.15; // Temporary final
        return 0.15;  // Standard final rule
      }
      if (normalizedAction.includes('correction')) return 0.10;
      if (normalizedAction.includes('withdrawal')) return 0.10;
      if (normalizedAction.includes('delay')) return 0.10;
      if (normalizedAction.includes('affirmation')) return 0.10;
      return 0.10;  // Other RULE subtypes
      
    case 'PRORULE':
      if (normalizedAction.includes('proposed rule')) {
        if (normalizedAction.includes('supplemental')) return 0.08;
        return 0.12;  // Standard proposed rule
      }
      if (normalizedAction.includes('advance notice') || 
          normalizedAction.includes('anprm')) return 0.08;
      if (normalizedAction.includes('request for information') ||
          normalizedAction.includes('rfi')) return 0.08;
      if (normalizedAction.includes('extension')) return 0.03;
      if (normalizedAction.includes('correction')) return 0.03;
      if (normalizedAction.includes('reopening')) return 0.03;
      return 0.08;  // Other PRORULE subtypes
      
    case 'NOTICE':
      // Tier 1 - Always evaluated
      if (normalizedAction.includes('coverage determination')) return 0.12;
      if (normalizedAction.includes('scheduling action')) return 0.12;
      if (normalizedAction.includes('civil money penalty')) return 0.12;
      if (normalizedAction.includes('exclusion')) return 0.12;
      if (normalizedAction.includes('final guidance')) return 0.12;
      
      // Tier 2 - Evaluated if from relevant agency
      if (normalizedAction.includes('draft guidance')) return 0.06;
      if (normalizedAction.includes('determination')) return 0.06;
      if (normalizedAction.includes('request for comments')) return 0.06;
      if (normalizedAction.includes('intent')) return 0.06;
      if (normalizedAction.includes('information collection')) return 0.06;
      if (normalizedAction.includes('petition')) return 0.06;
      
      // Tier 3 - Always excluded (S3 Rule 4)
      if (normalizedAction.includes('meeting')) return 0.00;
      if (normalizedAction.includes('nomination')) return 0.00;
      if (normalizedAction.includes('grant')) return 0.00;
      if (normalizedAction.includes('funding opportunity')) return 0.00;
      if (normalizedAction.includes('delegation')) return 0.00;
      if (normalizedAction.includes('privacy act')) return 0.00;
      if (normalizedAction.includes('system of records')) return 0.00;
      
      return 0.06;  // Other NOTICE subtypes
      
    case 'PRESDOCU':
      const subtype = extractPresidentialSubtype(action);
      if (subtype === 'executive_order') return 0.12;
      if (subtype === 'memorandum') return 0.08;
      if (subtype === 'determination') return 0.02;
      if (subtype === 'notice') return 0.02;
      if (subtype === 'proclamation') return 0.02;
      return 0.02;
      
    default:
      return 0.00;
  }
}

// S3 Rule 4: Meeting/grant exclusion floor
function shouldExcludeByDocumentType(
  type: string,
  action: string,
  kScore: number = 0
): boolean {
  const dScore = calculateDocumentTypeScore(type as any, action);
  if (dScore === 0.00 && kScore < 0.40) {
    return true;  // Exclude meetings, grants, etc. unless strong keywords
  }
  return false;
}
```

### 3C. Bonus Modifiers (B Score)

```typescript
interface BonusModifiers {
  significant_rule: boolean;      // +0.05
  multiple_healthcare_agencies: boolean;  // +0.05
  health_section: boolean;        // +0.03
  cfr_match: boolean;            // +0.10
  active_comment_period: boolean; // +0.02
}

async function calculateBonusModifiers(
  entity: any,
  stageOutput: StageOutput
): Promise<number> {
  let bonus = 0;
  const modifiers: Partial<BonusModifiers> = {};
  
  // Significant rule (EO 12866)
  if (entity.significant === true) {
    bonus += 0.05;
    modifiers.significant_rule = true;
  }
  
  // Multiple healthcare agencies
  if (entity.agencies && Array.isArray(entity.agencies)) {
    const healthcareAgencies = [
      'food-and-drug-administration',
      'centers-for-medicare-medicaid-services',
      'drug-enforcement-administration',
      'health-and-human-services-department'
    ];
    const matchCount = entity.agencies.filter((a: any) => 
      healthcareAgencies.some(ha => a.slug?.includes(ha))
    ).length;
    
    if (matchCount >= 2) {
      bonus += 0.05;
      modifiers.multiple_healthcare_agencies = true;
    }
  }
  
  // Health-and-public-welfare section
  if (entity.sections?.includes('health-and-public-welfare')) {
    bonus += 0.03;
    modifiers.health_section = true;
  }
  
  // CFR reference to relevant title/part (connects to Stage 1)
  if (stageOutput.classifications.some(c => c.assigned_by === 'structural')) {
    bonus += 0.10;
    modifiers.cfr_match = true;
  }
  
  // Active comment period
  if (entity.comments_close_on) {
    const closeDate = new Date(entity.comments_close_on);
    if (closeDate > new Date()) {
      bonus += 0.02;
      modifiers.active_comment_period = true;
    }
  }
  
  // Store modifier details for audit
  stageOutput.partial_signals.push({
    type: 'bonus_modifiers',
    details: modifiers
  });
  
  return Math.min(bonus, 0.10);  // Cap at 0.10
}
```

### 3D. Partial Score and Early Resolution

```typescript
async function stage2Metadata(
  entity_id: string,
  ruleSet: RuleSet,
  input: StageOutput
): Promise<StageOutput> {
  const entity = await loadEntity(entity_id);
  let output = { ...input };
  output.stage = 'metadata';
  
  // Calculate A score (agency)
  const agencyScore = await calculateAgencyScore(entity, ruleSet, output);
  output.score_components.A = agencyScore.score;
  
  // Handle excluded agencies (S3 Step 2)
  if (agencyScore.excluded) {
    output.partial_signals.push({
      type: 'excluded_agency',
      details: { agency: agencyScore.agency_slug }
    });
  }
  
  // Calculate D score (document type)
  if (entity.source_type === 'federal_register') {
    output.score_components.D = calculateDocumentTypeScore(
      entity.type,
      entity.action || ''
    );
  }
  
  // Calculate B score (bonus modifiers)
  output.score_components.B = await calculateBonusModifiers(entity, output);
  
  // Update composite score (partial - no K yet)
  const partialScore = output.score_components.A + 
                      output.score_components.D + 
                      output.score_components.B;
  
  // Early resolution check
  if (partialScore >= 0.60 && !agencyScore.excluded && agencyScore.single_domain) {
    output.composite_score = partialScore;
    output.s3_tier = 'relevant';
    output.classifications = [{
      domain_code: agencyScore.primary_domain,
      confidence: partialScore,
      is_primary: true,
      assigned_by: 'agency',
      rule_id: agencyScore.rule_id
    }];
    
    if (agencyScore.secondary_domains) {
      for (const domain of agencyScore.secondary_domains) {
        output.classifications.push({
          domain_code: domain,
          confidence: partialScore * 0.8,  // Secondary domains get lower confidence
          is_primary: false,
          assigned_by: 'agency',
          rule_id: agencyScore.rule_id
        });
      }
    }
  }
  
  return output;
}

// Supporting function for agency scoring
async function calculateAgencyScore(
  entity: any,
  ruleSet: RuleSet,
  output: StageOutput
): Promise<{
  score: number;
  excluded: boolean;
  single_domain: boolean;
  primary_domain?: string;
  secondary_domains?: string[];
  agency_slug?: string;
  rule_id?: string;
}> {
  // Check for agency in metadata
  const agencySlug = entity.agencies?.[0]?.slug || 
                    entity.agency_names?.[0]?.toLowerCase().replace(/\s+/g, '-');
  
  if (!agencySlug) {
    return { score: 0.15, excluded: false, single_domain: false };  // Unknown agency
  }
  
  // Check exclusion list first
  if (ruleSet.exclusion_rules.includes(agencySlug)) {
    return { 
      score: 0, 
      excluded: true, 
      single_domain: false,
      agency_slug: agencySlug
    };
  }
  
  // Find agency rule
  const agencyRule = ruleSet.agency_rules.find(rule => 
    rule.agency_slug === agencySlug
  );
  
  if (!agencyRule) {
    return { score: 0.15, excluded: false, single_domain: false };  // Unknown = 0.15
  }
  
  return {
    score: agencyRule.a_score,
    excluded: false,
    single_domain: !agencyRule.secondary_domains || agencyRule.secondary_domains.length === 0,
    primary_domain: agencyRule.primary_domain,
    secondary_domains: agencyRule.secondary_domains,
    agency_slug: agencySlug,
    rule_id: agencyRule.rule_id
  };
}
```
