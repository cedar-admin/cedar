## 2. Taxonomy Context Management

### 2A. Pruning Strategy

```typescript
function selectTaxonomyBranches(
  partialSignals: PartialSignal[],
  entityMetadata: EntityMetadata,
  fullTaxonomy: TaxonomyNode[]
): TaxonomyNode[] {
  const selectedBranches = new Set<string>();
  const branchScores = new Map<string, number>();
  
  // 1. Add branches from partial signals
  for (const signal of partialSignals) {
    if (signal.type === 'structural_match') {
      selectedBranches.add(signal.l1_domain);
      branchScores.set(signal.l1_domain, 1.0);
    } else if (signal.type === 'agency_match') {
      // From S8-C agency mapping
      const agencyDomains = getAgencyDomains(signal.agency_slug);
      agencyDomains.forEach(d => {
        selectedBranches.add(d.l1_domain);
        branchScores.set(d.l1_domain, d.confidence);
      });
    } else if (signal.type === 'keyword_match') {
      // Keywords may suggest domains
      const keywordDomains = getKeywordDomains(signal.keywords);
      keywordDomains.forEach(d => {
        selectedBranches.add(d.l1_domain);
        const current = branchScores.get(d.l1_domain) || 0;
        branchScores.set(d.l1_domain, Math.max(current, d.confidence));
      });
    }
  }
  
  // 2. Ensure minimum 3 branches (prevent tunnel vision)
  if (selectedBranches.size < 3) {
    // Add high-relevance branches based on entity type
    const candidateBranches = rankBranchesByRelevance(
      entityMetadata,
      Array.from(fullTaxonomy.values())
    );
    
    for (const branch of candidateBranches) {
      if (selectedBranches.size >= 3) break;
      selectedBranches.add(branch.code);
    }
  }
  
  // 3. Build pruned taxonomy
  const prunedTaxonomy: TaxonomyNode[] = [];
  
  for (const l1Code of selectedBranches) {
    const l1Node = fullTaxonomy.find(n => n.code === l1Code);
    if (!l1Node) continue;
    
    const score = branchScores.get(l1Code) || 0.5;
    
    if (score >= 0.7) {
      // High confidence: include full branch (L1 + all L2 + L3 codes with descriptions)
      prunedTaxonomy.push({
        ...l1Node,
        includeDepth: 'L3_full',
        children: includeChildrenToDepth(l1Node, 3, true)
      });
    } else {
      // Lower confidence: include L1 + L2 with descriptions, L3 codes only
      prunedTaxonomy.push({
        ...l1Node,
        includeDepth: 'L3_codes',
        children: includeChildrenToDepth(l1Node, 3, false)
      });
    }
  }
  
  // 4. Add remaining L1 branches as summaries (escape hatch)
  const includedL1s = new Set(prunedTaxonomy.map(n => n.code));
  for (const l1Node of fullTaxonomy) {
    if (!includedL1s.has(l1Node.code)) {
      prunedTaxonomy.push({
        code: l1Node.code,
        name: l1Node.name,
        description: l1Node.description.substring(0, 100) + '...',
        includeDepth: 'summary_only'
      });
    }
  }
  
  return prunedTaxonomy;
}

function rankBranchesByRelevance(
  metadata: EntityMetadata,
  branches: TaxonomyNode[]
): TaxonomyNode[] {
  // Heuristic ranking based on source type and agency
  const rankings = branches.map(branch => ({
    branch,
    score: calculateBranchRelevance(branch, metadata)
  }));
  
  return rankings
    .sort((a, b) => b.score - a.score)
    .map(r => r.branch);
}
```

Selection criteria for the 3 minimum branches when no signals exist:
1. **fda-regulation** (covers drugs, devices, biologics - high volume)
2. **controlled-substances** (high risk, frequently referenced)
3. **medicare-billing** OR **compounding** (based on document type)

Token budget estimates:
- Full branch (L1-L3 with descriptions): ~800-1000 tokens
- Partial branch (L1-L2 + L3 codes): ~400-600 tokens
- Summary branch (L1 only): ~50 tokens
- Target total: 2500-3000 tokens for typical 3-5 branch selection

### 3C. Batch Grouping

```typescript
function groupEntitiesForBatching(
  entities: QueuedEntity[]
): EntityBatch[] {
  const groups = new Map<string, QueuedEntity[]>();
  
  for (const entity of entities) {
    // Group key based on partial signals
    const key = generateGroupKey(entity);
    if (!groups.has(key)) {
      groups.set(key, []);
    }
    groups.get(key)!.push(entity);
  }
  
  // Convert groups to batches of size 3
  const batches: EntityBatch[] = [];
  for (const [key, groupEntities] of groups) {
    for (let i = 0; i < groupEntities.length; i += 3) {
      batches.push({
        entities: groupEntities.slice(i, i + 3),
        groupKey: key,
        sharedTaxonomy: selectTaxonomyBranches(
          groupEntities[i].partialSignals,
          groupEntities[i].metadata,
          fullTaxonomy
        )
      });
    }
  }
  
  return batches;
}

function generateGroupKey(entity: QueuedEntity): string {
  // Entities share taxonomy context if they have similar signals
  const signals = entity.partialSignals;
  const l1Domains = signals
    .filter(s => s.l1_domain)
    .map(s => s.l1_domain)
    .sort()
    .join(',');
  
  const agency = signals.find(s => s.type === 'agency_match')?.agency_slug || 'unknown';
  
  return `${l1Domains}:${agency}`;
}
```

Expected cost savings: 40-50% reduction in API costs through batching and taxonomy sharing.

### 3D. Error Handling

```typescript
async function callClassificationAPI(
  batch: EntityBatch,
  config: ClassificationConfig
): Promise<ClassificationResult[]> {
  let lastError: Error | null = null;
  
  for (let attempt = 1; attempt <= config.max_retries; attempt++) {
    try {
      const response = await anthropic.messages.create({
        model: config.stage4_model,
        max_tokens: 400 * batch.entities.length,
        temperature: 0.3,
        system: systemPrompt,
        messages: [{ role: 'user', content: userMessage }]
      });
      
      // Parse response
      const results = parseClassificationResponse(response.content);
      
      // Validate
      for (const result of results) {
        validateClassificationResult(result, batch.sharedTaxonomy);
      }
      
      return results;
      
    } catch (error) {
      lastError = error;
      
      if (error.code === 'rate_limit_error') {
        // Exponential backoff: 2^attempt seconds
        await sleep(Math.pow(2, attempt) * 1000);
        continue;
      }
      
      if (error.code === 'invalid_request_error' && batch.entities.length > 1) {
        // Reduce batch size and retry individually
        const individualResults = [];
        for (const entity of batch.entities) {
          const singleResult = await callClassificationAPI(
            { ...batch, entities: [entity] },
            config
          );
          individualResults.push(...singleResult);
        }
        return individualResults;
      }
      
      // Non-retryable error
      throw error;
    }
  }
  
  throw new Error(`Classification failed after ${config.max_retries} attempts`, { cause: lastError });
}

function validateClassificationResult(
  result: any,
  validTaxonomy: TaxonomyNode[]
): void {
  // Check domain codes exist in provided taxonomy
  const validCodes = extractAllCodes(validTaxonomy);
  
  for (const classification of result.classifications) {
    if (!validCodes.has(classification.domain_code)) {
      throw new Error(`Invalid domain code: ${classification.domain_code}`);
    }
    
    if (classification.confidence < 0 || classification.confidence > 1) {
      throw new Error(`Invalid confidence: ${classification.confidence}`);
    }
  }
  
  // Check primary domain constraint
  const primaryCount = result.classifications.filter(c => c.is_primary).length;
  if (primaryCount !== 1 && result.classifications.length > 0) {
    throw new Error(`Exactly one primary classification required, found ${primaryCount}`);
  }
}
```

## 4. Confidence Tiers and HITL Integration

### 4A. Confidence Tiers

| Tier | Range | Action | HITL Behavior |
|---|---|---|---|
| High | ≥ 0.75 | Auto-accept | Sampled for quality (5% sample rate) |
| Medium | 0.60 to 0.74 | Accept with flag | Queued for eventual review |
| Low | < 0.60 | Needs review | Not authoritative until reviewed |

Rationale:
- S3's threshold of 0.60 represents "clearly relevant" — AI should be more confident
- 0.75 threshold aligns with "strong match" in prompt instructions
- Initial deployment should monitor accuracy at each tier and adjust after 500 entities

Calibration process:
1. Deploy with thresholds above
2. Review first 100 entities in each tier
3. Calculate accuracy: (correct classifications / total reviewed)
4. Adjust thresholds to achieve 95% accuracy in High tier, 85% in Medium

### 4B. HITL Queue Integration

```typescript
interface HITLQueueEntry {
  entity_id: string;
  classification_result: ClassificationResult;
  priority_score: number;
  queue_reason: 'low_confidence' | 'qa_sample' | 'confusion_risk' | 'correction_pattern';
  created_at: string;
  metadata: {
    stage_outputs: StageOutput[];
    prompt_version: string;
    model_used: string;
    confusion_domains?: string[];  // If matched a known confusion pattern
  };
}

function calculatePriorityScore(
  result: ClassificationResult,
  entity: Entity
): number {
  const weights = {
    confidence: -0.4,      // Lower confidence = higher priority
    impact: 0.3,          // More affected practices = higher priority
    confusion_risk: 0.2,  // Known confusion pair = higher priority
    recency: 0.1         // Newer entities slightly higher priority
  };
  
  // Impact: count practices where primary domain has Medium+ relevance
  const primaryDomain = result.classifications.find(c => c.is_primary);
  const impactScore = primaryDomain 
    ? countPracticesWithRelevance(primaryDomain.domain_code, 'medium') / 14
    : 0;
  
  // Confusion risk: check if domains are in known confusion pairs
  const confusionScore = hasConfusionRisk(result.classifications) ? 1 : 0;
  
  // Recency: normalize to 0-1 (last 30 days)
  const ageInDays = (Date.now() - new Date(entity.created_at).getTime()) / (1000 * 60 * 60 * 24);
  const recencyScore = Math.max(0, 1 - (ageInDays / 30));
  
  return (
    weights.confidence * (1 - result.classifications[0].confidence) +
    weights.impact * impactScore +
    weights.confusion_risk * confusionScore +
    weights.recency * recencyScore
  );
}
```

```typescript
  const accuracyByTier = {
    high: samples.filter(s => s.original_confidence >= 0.75),
    medium: samples.filter(s => s.original_confidence >= 0.60 && s.original_confidence < 0.75),
    low: samples.filter(s => s.original_confidence < 0.60)
  };
  
  const report = {
    high_tier_accuracy: calculateAccuracy(accuracyByTier.high),
    recommended_adjustments: []
  };
  
  // If high tier accuracy < 95%, recommend raising threshold
  if (report.high_tier_accuracy < 0.95) {
    report.recommended_adjustments.push({
      parameter: 'high_confidence_threshold',
      current: config.high_confidence_threshold,
      recommended: config.high_confidence_threshold + 0.05
    });
  }
  
  return report;
}
```

### 5D. Active Learning

```typescript
class ActiveLearningSelector {
  async selectHighValueTargets(
    unreviewed: ClassificationResult[],
    recentCorrections: ClassificationCorrection[]
  ): Promise<string[]> {
    const scores = new Map<string, number>();
    
    for (const result of unreviewed) {
      let score = 0;
      
      // 1. Near decision boundaries (confidence close to thresholds)
      const confidence = result.classifications[0].confidence;
      const distanceToThreshold = Math.min(
        Math.abs(confidence - 0.60),
        Math.abs(confidence - 0.75)
      );
      score += (0.15 - distanceToThreshold) * 10; // Max 10 points
      
      // 2. In high-confusion domains
      const confusionScore = await this.getConfusionScore(
        result.classifications[0].domain_code,
        recentCorrections
      );
      score += confusionScore * 5; // Max 5 points
      
      // 3. Underrepresented domains (few recent classifications)
      const domainFrequency = await this.getDomainFrequency(
        result.classifications[0].domain_code
      );
      score += (1 - domainFrequency) * 3; // Max 3 points
      
      // 4. Multiple secondary domains (complex classification)
      if (result.classifications.length > 2) {
        score += 2;
      }
      
      scores.set(result.entity_id, score);
    }
    
    // Return top 10% for review
    return Array.from(scores.entries())
      .sort((a, b) => b[1] - a[1])
      .slice(0, Math.ceil(unreviewed.length * 0.1))
      .map(([entityId]) => entityId);
  }
  
  async recalibrateThresholds(corrections: ClassificationCorrection[]): Promise<ThresholdAdjustment[]> {
    // After 100 corrections, analyze if thresholds need adjustment
    if (corrections.length < 100) return [];
    
    const accuracyByConfidence = await db.query(`
      SELECT
        CASE 
          WHEN confidence >= 0.75 THEN 'high'
          WHEN confidence >= 0.60 THEN 'medium'
          ELSE 'low'
        END as tier,
        COUNT(*) as total,
        SUM(CASE WHEN corrected_from IS NOT NULL THEN 1 ELSE 0 END) as corrected,
        1.0 - (SUM(CASE WHEN corrected_from IS NOT NULL THEN 1 ELSE 0 END)::float / COUNT(*)) as accuracy
      FROM kg_classification_log
      WHERE stage = 4
        AND created_at > NOW() - INTERVAL '7 days'
      GROUP BY tier
    `);
    
    const adjustments: ThresholdAdjustment[] = [];
    
    for (const row of accuracyByConfidence) {
      if (row.tier === 'high' && row.accuracy < 0.95) {
        adjustments.push({
          parameter: 'high_confidence_threshold',
          current: 0.75,
          recommended: 0.80,
          reason: `High tier accuracy ${row.accuracy} below 95% target`
        });
      }
    }
    
    return adjustments;
  }
}
```

```typescript
          await classifyAsIrrelevant(result.entity_id, result.confidence, result.brief_reason);
        } else if (result.relevant === true) {
          // Actually relevant - send back to Stage 4 with expanded taxonomy
          await inngest.send({
            name: 'cedar/classify.stage4',
            data: {
              entity_ids: [result.entity_id],
              reason: 'stage5_found_relevant',
              expand_all_branches: true
            }
          });
        } else {
          // Uncertain - flag for manual review
          await handleUncertainEntities([{
            entity_id: result.entity_id,
            uncertainty_type: 'conflicting_signals',
            stage5_result: result,
            priority: 0.3
          }]);
        }
      }
    });
    
    return {
      processed: results.length,
      irrelevant: results.filter(r => r.relevant === false).length,
      reclassify: results.filter(r => r.relevant === true).length,
      uncertain: results.filter(r => r.confidence < config.irrelevance_confidence_threshold).length
    };
  }
);

// Helper functions
function determineS3Tier(result: ClassificationResult): string {
  const confidence = result.classifications[0]?.confidence || 0;
  
  if (confidence >= 0.60) return 'relevant';
  if (confidence >= 0.40) return 'review';
  if (confidence >= 0.20) return 'low_priority';
  return 'irrelevant';
}

async function prepareClassificationPrompts(
  entities: Entity[],
  taxonomy: TaxonomyNode[],
  config: ClassificationConfig
): Promise<{ systemPrompt: string; userPrompt: string }> {
  // Load prompt template
  const template = await loadPromptTemplate(config.prompt_version, 'stage4_system');
  
  // Format taxonomy context
  const taxonomyContext = formatTaxonomyContext(taxonomy);
  
  // Get cross-classification examples
  const crossExamples = await getCrossClassificationExamples(config.s7c_examples_count);
  
  // Build system prompt
  const systemPrompt = template
    .replace('{{taxonomy_context}}', taxonomyContext)
    .replace('{{cross_classification_examples}}', crossExamples)
    .replace('{{partial_signals_context}}', formatPartialSignalsContext(entities[0]))
    .replace('{{output_schema}}', OUTPUT_SCHEMA_JSON);
  
  // Build user prompt for batch
  const userPrompt = entities.map(entity => 
    formatEntityForClassification(entity)
  ).join('\n\n---\n\n');
  
  return { systemPrompt, userPrompt };
}
```
