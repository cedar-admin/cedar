### 2A. Implement S3's K Score Calculation

```typescript
interface KeywordMatch {
  phrase: string;
  domain_code: string;
  signal_strength: 'strong' | 'moderate';
  homonym_risk: boolean;
  disambiguation_confirmed: boolean;
  match_location: 'title' | 'abstract' | 'both';
}

interface DomainAssignment {
  domain_code: string;
  confidence: number;
  contributing_keywords: string[];
  is_primary: boolean;
}

function calculateKScore(
  matches: KeywordMatch[],
  entity_text: string
): { 
  k_score: number; 
  matched_domains: DomainAssignment[]; 
  match_details: KeywordMatch[] 
} {
  // Group matches by domain
  const domainMatches = new Map<string, KeywordMatch[]>();
  
  for (const match of matches) {
    const domain = match.domain_code;
    if (!domainMatches.has(domain)) {
      domainMatches.set(domain, []);
    }
    domainMatches.get(domain)!.push(match);
  }
  
  // Calculate per-domain scores and overall K score
  let maxKScore = 0;
  const domainScores: Array<{domain: string, score: number, keywords: string[]}> = [];
  
  for (const [domain, domainKeywords] of domainMatches) {
    let domainKScore = 0;
    const contributingKeywords: string[] = [];
    
    // Count strong and moderate matches
    const strongMatches = domainKeywords.filter(m => 
      m.signal_strength === 'strong' && 
      (!m.homonym_risk || m.disambiguation_confirmed)
    );
    const moderateMatches = domainKeywords.filter(m => 
      m.signal_strength === 'moderate' && 
      (!m.homonym_risk || m.disambiguation_confirmed)
    );
    const unconfirmedHomonyms = domainKeywords.filter(m => 
      m.homonym_risk && !m.disambiguation_confirmed
    );
    
    // Apply S3's scoring rules
    if (strongMatches.length >= 2) {
      domainKScore = 0.40;
      contributingKeywords.push(...strongMatches.map(m => m.phrase));
    } else if (strongMatches.length === 1 && moderateMatches.length >= 1) {
      domainKScore = 0.35;
      contributingKeywords.push(strongMatches[0].phrase);
      contributingKeywords.push(moderateMatches[0].phrase);
    } else if (strongMatches.length === 1) {
      domainKScore = 0.30;
      contributingKeywords.push(strongMatches[0].phrase);
    } else if (moderateMatches.length >= 2) {
      domainKScore = 0.25;
      contributingKeywords.push(...moderateMatches.slice(0, 2).map(m => m.phrase));
    } else if (moderateMatches.length === 1) {
      domainKScore = 0.15;
      contributingKeywords.push(moderateMatches[0].phrase);
    } else if (unconfirmedHomonyms.length > 0) {
      domainKScore = 0.05;
      contributingKeywords.push(unconfirmedHomonyms[0].phrase);
    }
    
    // Check for cross-domain specialty phrases (always 0.35)
    const crossDomainMatch = domainKeywords.find(m => 
      CROSS_DOMAIN_PHRASES.includes(m.phrase)
    );
    if (crossDomainMatch && domainKScore < 0.35) {
      domainKScore = 0.35;
      contributingKeywords.length = 0;
      contributingKeywords.push(crossDomainMatch.phrase);
    }
    
    if (domainKScore > 0) {
      domainScores.push({
        domain,
        score: domainKScore,
        keywords: contributingKeywords
      });
    }
    
    maxKScore = Math.max(maxKScore, domainKScore);
  }
  
  // Sort domains by score (highest first)
  domainScores.sort((a, b) => b.score - a.score);
  
  // Convert to DomainAssignment format
  const matched_domains: DomainAssignment[] = domainScores.map((ds, index) => ({
    domain_code: ds.domain,
    confidence: ds.score,
    contributing_keywords: ds.keywords,
    is_primary: index === 0
  }));
  
  return {
    k_score: maxKScore,  // Overall K score is the highest domain score
    matched_domains,
    match_details: matches
  };
}
```

### 2B. Text Matching Patterns

```typescript
interface TextNormalizationConfig {
  lowercase: boolean;
  strip_punctuation: boolean;
  normalize_whitespace: boolean;
  expand_abbreviations: boolean;
  stem_plurals: boolean;
}

const DEFAULT_NORMALIZATION: TextNormalizationConfig = {
  lowercase: true,
  strip_punctuation: true,
  normalize_whitespace: true,
  expand_abbreviations: false,  // Keep precise - S3 phrases are specific
  stem_plurals: true
};

function normalizeText(text: string, config: TextNormalizationConfig): string {
  let normalized = text;
  
  if (config.lowercase) {
    normalized = normalized.toLowerCase();
  }
  
  if (config.strip_punctuation) {
    // Keep hyphens and parentheses that are part of terms like "503(k)" or "anti-kickback"
    normalized = normalized.replace(/[^\w\s\-()]/g, ' ');
  }
  
  if (config.normalize_whitespace) {
    normalized = normalized.replace(/\s+/g, ' ').trim();
  }
  
  if (config.stem_plurals) {
    // Simple plural handling for common medical terms
    normalized = normalized
      .replace(/\bsubstances\b/g, 'substance')
      .replace(/\bservices\b/g, 'service')
      .replace(/\brequirements\b/g, 'requirement')
      .replace(/\bpractitioners\b/g, 'practitioner');
  }
  
  return normalized;
}

interface MatchingAlgorithm {
  // Pre-compile phrases into a single regex for performance
  compilePhrases(phrases: KeywordPhrase[]): RegExp;
  
  // Match against concatenated title + abstract
  findMatches(
    entityText: string, 
    compiledRegex: RegExp,
    phraseMap: Map<string, KeywordPhrase>
  ): KeywordMatch[];
}

class PhraseMatchingEngine implements MatchingAlgorithm {
  compilePhrases(phrases: KeywordPhrase[]): RegExp {
    // Sort phrases by length (longest first) to match multi-word phrases before single words
    const sortedPhrases = phrases
      .map(p => p.phrase)
      .sort((a, b) => b.split(' ').length - a.split(' ').length);
    
    // Escape special regex characters but preserve word boundaries
    const escapedPhrases = sortedPhrases.map(phrase => 
      phrase.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')
        .split(' ')
        .join('\\s+')  // Allow flexible whitespace
    );
    
    // Create a single regex with word boundaries
    const pattern = `\\b(${escapedPhrases.join('|')})\\b`;
    return new RegExp(pattern, 'gi');
  }
  
  findMatches(
    entityText: string,
    compiledRegex: RegExp,
    phraseMap: Map<string, KeywordPhrase>
  ): KeywordMatch[] {
    const normalizedText = normalizeText(entityText, DEFAULT_NORMALIZATION);
    const matches: KeywordMatch[] = [];
    const seen = new Set<string>();
    
    let match;
    while ((match = compiledRegex.exec(normalizedText)) !== null) {
      const matchedText = match[1].toLowerCase();
      
      // Find the original phrase (handling normalization differences)
      let originalPhrase: KeywordPhrase | undefined;
      for (const [phrase, data] of phraseMap) {
        if (normalizeText(phrase, DEFAULT_NORMALIZATION).includes(matchedText)) {
          originalPhrase = data;
          break;
        }
      }
      
      if (originalPhrase && !seen.has(originalPhrase.phrase)) {
        seen.add(originalPhrase.phrase);
        
        // Determine match location
        const titleNormalized = normalizeText(
          entityText.substring(0, 200), // Assume first 200 chars is title
          DEFAULT_NORMALIZATION
        );
        const matchLocation = titleNormalized.includes(matchedText) ? 'title' : 'abstract';
        
        matches.push({
          phrase: originalPhrase.phrase,
          domain_code: originalPhrase.primary_domain_code,
          signal_strength: originalPhrase.signal_strength,
          homonym_risk: originalPhrase.homonym_risk,
          disambiguation_confirmed: false, // Will be set by disambiguation engine
          match_location: matchLocation as 'title' | 'abstract'
        });
      }
    }
    
    return matches;
  }
}

// Performance estimate for 99K entities × 237 phrases:
// - Single compiled regex: ~5ms per entity
// - Total processing time: ~8-10 minutes single-threaded
// - With 10-way parallelization: ~1 minute
// - Memory usage: ~500MB for compiled patterns + entity text
```

### 2D. Domain Assignment from Keyword Matches

```typescript
interface DomainScoringConfig {
  stage1_2_boost: number;           // Boost for domains matching Stage 1-2 signals
  title_match_boost: number;        // Extra weight for title matches
  multi_domain_threshold: number;   // Min score for secondary domain assignment
  max_secondary_domains: number;    // Cap on secondary assignments
}

const DEFAULT_SCORING_CONFIG: DomainScoringConfig = {
  stage1_2_boost: 1.5,
  title_match_boost: 1.2,
  multi_domain_threshold: 0.10,
  max_secondary_domains: 4
};

function assignDomainsFromKeywords(
  keywordMatches: KeywordMatch[],
  stage1_2_domains: string[],  // Domains partially assigned by Stages 1-2
  config: DomainScoringConfig = DEFAULT_SCORING_CONFIG
): DomainAssignment[] {
  // Calculate weighted score for each domain
  const domainScores = new Map<string, {
    score: number;
    keywords: string[];
    titleMatches: number;
    strongMatches: number;
  }>();
  
  // Initialize scores for Stage 1-2 domains with boost
  for (const domain of stage1_2_domains) {
    domainScores.set(domain, {
      score: 0.05 * config.stage1_2_boost,  // Small baseline boost
      keywords: [],
      titleMatches: 0,
      strongMatches: 0
    });
  }
  
  // Accumulate keyword contributions
  for (const match of keywordMatches) {
    const domain = match.domain_code;
    
    if (!domainScores.has(domain)) {
      domainScores.set(domain, {
        score: 0,
        keywords: [],
        titleMatches: 0,
        strongMatches: 0
      });
    }
    
    const domainData = domainScores.get(domain)!;
    
    // Base contribution from keyword weight
    const cluster = DOMAIN_KEYWORD_CLUSTERS.find(c => c.domain_code === domain);
    const keywordWeight = cluster?.keywords.find(k => k.phrase === match.phrase)?.weight || 0.5;
    
    let contribution = keywordWeight * 0.1;  // Scale to reasonable range
    
    // Apply signal strength multiplier
    if (match.signal_strength === 'strong') {
      contribution *= 1.5;
      domainData.strongMatches++;
    }
    
    // Apply homonym penalty if not disambiguated
    if (match.homonym_risk && !match.disambiguation_confirmed) {
      contribution *= 0.3;
    }
    
    // Title match boost
    if (match.match_location === 'title' || match.match_location === 'both') {
      contribution *= config.title_match_boost;
      domainData.titleMatches++;
    }
    
    // Stage 1-2 alignment boost
    if (stage1_2_domains.includes(domain)) {
      contribution *= config.stage1_2_boost;
    }
    
    domainData.score += contribution;
    if (!domainData.keywords.includes(match.phrase)) {
      domainData.keywords.push(match.phrase);
    }
  }
  
  // Apply domain-specific thresholds and convert to assignments
  const assignments: DomainAssignment[] = [];
  const sortedDomains = Array.from(domainScores.entries())
    .sort((a, b) => b[1].score - a[1].score);
  
  for (const [domain, data] of sortedDomains) {
    const cluster = DOMAIN_KEYWORD_CLUSTERS.find(c => c.domain_code === domain);
    const threshold = cluster?.match_threshold || 0.15;
    
    // Check if score meets domain's threshold
    if (data.score >= threshold) {
      // Normalize confidence to 0-1 range
      const confidence = Math.min(1.0, data.score);
      
      assignments.push({
        domain_code: domain,
        confidence: confidence,
        contributing_keywords: data.keywords,
        is_primary: assignments.length === 0
      });
      
      // Stop adding secondary domains after limit
      if (assignments.length > config.max_secondary_domains) {
        break;
      }
    }
  }
  
  // Ensure at least primary domain for high-confidence matches
  if (assignments.length === 0 && sortedDomains.length > 0) {
    const [topDomain, topData] = sortedDomains[0];
    if (topData.score > 0) {
      assignments.push({
        domain_code: topDomain,
        confidence: Math.min(0.5, topData.score),  // Cap low-confidence primaries
        contributing_keywords: topData.keywords,
        is_primary: true
      });
    }
  }
  
  return assignments;
}
```

```typescript
function applyCrossClassification(
  entity_id: string,
  primary_classifications: DomainAssignment[],
  entity_text: string,
  applicable_triggers: CrossClassificationTrigger[]
): DomainAssignment[] {
  const additional_domains: DomainAssignment[] = [];
  const assigned_domains = new Set(primary_classifications.map(c => c.domain_code));
  
  for (const primary of primary_classifications) {
    // Find triggers where this domain is the source
    const relevant_triggers = applicable_triggers.filter(t => 
      t.source_domain === primary.domain_code ||
      (t.bidirectional && t.target_domain === primary.domain_code)
    );
    
    for (const trigger of relevant_triggers) {
      // Check if trigger keywords are present
      const normalized_text = normalizeText(entity_text, DEFAULT_NORMALIZATION);
      const keyword_match = trigger.trigger_keywords.some(keyword => 
        normalized_text.includes(keyword.toLowerCase())
      );
      
      if (keyword_match) {
        const target_domain = trigger.source_domain === primary.domain_code 
          ? trigger.target_domain 
          : trigger.source_domain;  // Bidirectional case
        
        // Skip if already assigned
        if (assigned_domains.has(target_domain)) {
          continue;
        }
        
        // Calculate secondary confidence
        const secondary_confidence = primary.confidence * trigger.confidence_discount;
        
        // Apply minimum confidence threshold
        if (secondary_confidence >= 0.50) {
          additional_domains.push({
            domain_code: target_domain,
            confidence: secondary_confidence,
            contributing_keywords: trigger.trigger_keywords.filter(k => 
              normalized_text.includes(k.toLowerCase())
            ),
            is_primary: false
          });
          
          assigned_domains.add(target_domain);
        }
      }
    }
  }
  
  // Sort by confidence and apply max limit
  additional_domains.sort((a, b) => b.confidence - a.confidence);
  return additional_domains.slice(0, 3);  // Max 3 secondary domains from cross-classification
}
```

### 4A. Operationalize S3's Homonym Risk Handling

```typescript
interface DisambiguationContext {
  agency_identifiers: string[];
  healthcare_terms: string[];
  regulatory_markers: string[];
  contradicting_terms: string[];  // Terms that suggest non-healthcare context
}

const DISAMBIGUATION_CONTEXT: DisambiguationContext = {
  agency_identifiers: [
    'FDA', 'DEA', 'CMS', 'OSHA', 'FTC', 'HHS', 'ONC', 'OCR', 'SAMHSA', 'AHRQ',
    'food and drug administration', 'drug enforcement administration',
    'centers for medicare', 'occupational safety', 'federal trade commission',
    'health and human services'
  ],
  healthcare_terms: [
    'patient', 'clinical', 'medical', 'healthcare', 'physician', 'pharmacy',
    'drug', 'device', 'hospital', 'practitioner', 'health plan', 'provider',
    'treatment', 'diagnosis', 'therapy', 'prescription', 'medication'
  ],
  regulatory_markers: [
    'CFR', 'USC', 'Public Law', 'final rule', 'proposed rule', 'guidance',
    'enforcement', 'compliance', 'requirement', 'regulation', 'standard'
  ],
  contradicting_terms: [
    'logistics', 'shipping schedule', 'manufacturing schedule', 'project scope',
    'financial audit', 'construction safety', 'software security', 'IT security',
    'general business', 'retail', 'consumer product' /* non-medical */
  ]
};

function disambiguate(
  keyword: KeywordPhrase,
  entity_text: string
): { confirmed: boolean; confidence_modifier: number } {
  // No disambiguation needed for non-homonym keywords
  if (!keyword.homonym_risk) {
    return { confirmed: true, confidence_modifier: 1.0 };
  }
  
  const normalized_text = normalizeText(entity_text, DEFAULT_NORMALIZATION);
  
  // Check for contradicting context first (strongest signal)
  const has_contradicting = DISAMBIGUATION_CONTEXT.contradicting_terms.some(term =>
    normalized_text.includes(term)
  );
  
  if (has_contradicting) {
    return { confirmed: false, confidence_modifier: 0.0 };
  }
  
  // Count positive context signals
  let context_score = 0;
  
  // Agency identifiers (strongest positive signal)
  const agency_matches = DISAMBIGUATION_CONTEXT.agency_identifiers.filter(term =>
    normalized_text.includes(term.toLowerCase())
  ).length;
  context_score += agency_matches * 2.0;
  
  // Healthcare terms (moderate signal)
  const healthcare_matches = DISAMBIGUATION_CONTEXT.healthcare_terms.filter(term =>
    normalized_text.includes(term.toLowerCase())
  ).length;
  context_score += healthcare_matches * 1.0;
  
  // Regulatory markers (weak signal)
  const regulatory_matches = DISAMBIGUATION_CONTEXT.regulatory_markers.filter(term =>
    normalized_text.includes(term.toLowerCase())
  ).length;
  context_score += regulatory_matches * 0.5;
  
  // Check keyword-specific required context
  if (keyword.disambiguation_requires && keyword.disambiguation_requires.length > 0) {
    const specific_match = keyword.disambiguation_requires.some(term =>
      normalized_text.includes(term.toLowerCase())
    );
    if (specific_match) {
      context_score += 3.0;  // Strong signal when specific requirement met
    }
  }
  
  // Determine confirmation based on context score
  if (context_score >= 3.0) {
    return { confirmed: true, confidence_modifier: 1.0 };
  } else if (context_score >= 1.5) {
    return { confirmed: true, confidence_modifier: 0.8 };
  } else if (context_score >= 0.5) {
    return { confirmed: false, confidence_modifier: 0.5 };
  } else {
    // Follow S3's rule: unconfirmed homonym gets 0.05 K contribution
    return { confirmed: false, confidence_modifier: 0.167 };  // 0.05/0.30 for Strong keyword
  }
}
```

### 4B. High-Risk Domain Handling

```typescript
const HIGH_RISK_DOMAINS = [
  'advertising-marketing',      // 8/14 keywords have homonym risk
  'business-operations.osha-compliance'  // 6/12 keywords have homonym risk
];

function applyHighRiskDomainRules(
  domain_assignments: DomainAssignment[],
  keyword_matches: KeywordMatch[],
  entity_agencies: string[]  // Agencies from Stage 2
): { assignments: DomainAssignment[]; needs_review: boolean } {
  let needs_review = false;
  
  for (const assignment of domain_assignments) {
    // Check if this is a high-risk domain
    const is_high_risk = HIGH_RISK_DOMAINS.some(hrd => 
      assignment.domain_code.startsWith(hrd)
    );
    
    if (is_high_risk && assignment.is_primary) {
      // Count confirming signals for this domain
      const domain_keywords = keyword_matches.filter(m => 
        m.domain_code === assignment.domain_code
      );
      
      const confirmed_keywords = domain_keywords.filter(m => 
        !m.homonym_risk || m.disambiguation_confirmed
      );
      
      // Check for relevant agency
      const has_relevant_agency = (
        (assignment.domain_code.startsWith('advertising-marketing') && 
         entity_agencies.includes('federal-trade-commission')) ||
        (assignment.domain_code.startsWith('business-operations.osha') && 
         entity_agencies.includes('occupational-safety-and-health-administration'))
      );
      
      // Apply S3's "two confirming signals" rule
      const confirming_signals = confirmed_keywords.length + (has_relevant_agency ? 1 : 0);
      
      if (confirming_signals < 2) {
        needs_review = true;
        // Reduce confidence for weak high-risk classifications
        assignment.confidence *= 0.8;
      }
    }
  }
  
  return { assignments: domain_assignments, needs_review };
}
```
